================================================================================
HOW TO USE THE KNOWLEDGE GRAPH
================================================================================

Loading the Graph:
  kg = load_graph_from_csv(
      'data/molecules_kg.csv',
      'data/proteins_kg.csv',
      'data/bioactivity_edges.csv',
      'data/similarity_edges.csv'
  )

Finding Reasoning Paths:
  paths = kg.find_paths('mol_12345', 'prot_67890', max_length=3)
  for path in paths:
      description = kg._describe_path(path)
      print(description)

Explaining Bioactivity:
  explanation = kg.explain_bioactivity('mol_12345', 'prot_67890')
  print(explanation['mechanisms'])

Making Predictions with Reasoning:
  result = kg.predict_with_reasoning('mol_12345', objective='bioactivity')
  print(f"Prediction: {result['prediction']}")
  print(f"Reasoning: {result['reasoning']}")

Finding Similar Molecules:
  similar = kg._find_similar_molecules('mol_12345')
  for mol in similar:
      print(f"Molecule {mol['molecule']}: similarity {mol['similarity']:.2f}")
